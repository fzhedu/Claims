94  uint ChangeSign() {
95  Int<value_size> temp;
96 
97  temp.SetMin();
98 
99  /*
100  if the value is equal that one which has been returned from SetMin
101  that means we can't change sign because the value is too big (bigger about one)
102 
103  e.g. when value_size = 1 and value is -2147483648 we can't change it to the
104  2147483648 because the max value which can be held is 2147483647
105 
106  we don't change the value and we're using this fact somewhere in some methods
107  (if we look on our value without the sign we get the correct value
108  eg. -2147483648 in Int<1> will be 2147483648 on the UInt<1> type)
109  */
110  if (operator==(temp))
111  return 1;
112 
113  temp.SetZero();
114  temp.UInt<value_size>::Sub(*this);
115 
116  operator=(temp);
117 
118  return 0;
119  }

316  uint Mul(Int<value_size> ss2) {
317  bool ss1_is_sign, ss2_is_sign;
318 
319  ss1_is_sign = IsSign();
320  ss2_is_sign = ss2.IsSign();
321 
322  /*
323  we don't have to check the carry from Abs (values will be correct
324  because next we're using the method Mul from the base class UInt
325  which is without a sign)
326  */
327  Abs();
328  ss2.Abs();
329 
330  if (UInt<value_size>::Mul(ss2))
331  return 1;
332 
333  /*
334  we have to examine the sign of the result now
335  but if the result is with the sign then:
336  1. if the signs were the same that means the result is too big
337  (the result must be without a sign)
338  2. if the signs were different that means if the result
339  is different from that one which has been returned from SetMin()
340  that is carry (result too big) but if the result is equal SetMin()
341  there'll be ok (and the next SetSign will has no effect because
342  the value is actually negative -- look at description of that case
343  in ChangeSign())
344  */
345  if (IsSign()) {
346  /*
347  there can be one case where signs are different and
348  the result will be equal the value from SetMin()
349  (this situation is ok)
350  */
351  if (ss1_is_sign != ss2_is_sign) {
352  Int<value_size> temp;
353  temp.SetMin();
354 
355  if (operator!=(temp))
356  /*
357  the result is too big
358  */
359  return 1;
360  } else {
361  /*
362  the result is too big
363  */
364  return 1;
365  }
366  }
367

369  SetSign();

474  bool was_sign = IsSign();
475  uint c = 0;
476 
477  if (was_sign)
478  c += Abs();
479 
480  uint c_temp = UInt<value_size>::Pow(pow);
